Código CSS utilizado:

/* Bloquear selección en todos los elementos */ *, *::before, *::after { -webkit-user-select: none; /* Safari, Chrome, Edge */ -moz-user-select: none; /* Firefox */ -ms-user-select: none; /* IE/Edge antiguo */ user-select: none; /* Estándar */ } /* Excepción: inputs y textareas */ input, textarea { -webkit-user-select: text; -moz-user-select: text; -ms-user-select: text; user-select: text; } /* Excepción: elementos específicos */ .permitir-seleccion { -webkit-user-select: text; -moz-user-select: text; -ms-user-select: text; user-select: text; }
